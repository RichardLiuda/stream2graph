# =============================================================================
# stream2graph — 前端依赖与环境说明（Node.js / pnpm）
#
# 注意：本文件不是 Python pip 用的 requirements.txt。
#       前端依赖由根目录与各子包的 package.json 声明，通过 pnpm 安装。
# =============================================================================

# -----------------------------------------------------------------------------
# 环境要求
# -----------------------------------------------------------------------------
# Node.js     : 建议使用当前 LTS（如 20.x），需满足 Next.js 15 运行要求
# 包管理器    : pnpm，版本须与根目录 package.json 中 "packageManager" 一致
#               （当前为 pnpm@10.30.3）
# -----------------------------------------------------------------------------

# -----------------------------------------------------------------------------
# 安装命令（在仓库根目录执行）
# -----------------------------------------------------------------------------
# 安装全部 workspace 前端相关依赖（含 apps/web、packages/*）
pnpm install

# -----------------------------------------------------------------------------
# 常用脚本（在仓库根目录执行）
# -----------------------------------------------------------------------------
# 启动 Next 开发服务器
pnpm run dev:web

# 类型检查（web）
pnpm run typecheck:web

# 构建前端
pnpm run build:web

# Lint（web）
pnpm run lint:web

# -----------------------------------------------------------------------------
# Workspace 包（与 pnpm-workspace.yaml 对应）
# -----------------------------------------------------------------------------
# apps/web                 @stream2graph/web        — Next.js 主站
# packages/ui              @stream2graph/ui         — 共享 UI 组件
# packages/contracts       @stream2graph/contracts  — 共享契约/类型
